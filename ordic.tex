% =======================
\section*{Core Vibe / Design}
\begin{itemize}
  \item \textbf{No gender.} Meaning is carried by \textbf{cases} and \textbf{particles}.
  \item \textbf{German-like syntax:} V2 in main clauses; verb-final in subordinates.
  \item \textbf{Spanish-like TAM richness:} expressed with \textbf{particles} (not heavy conjugation).
  \item \textbf{Emotion is grammatical:} emotion particles can modify intent/truth conditions.
\end{itemize}

% =======================
\section*{Phonology \& Orthography (default)}
\begin{itemize}
  \item \textbf{Vowels:} a e i o u (optional: ä ö)
  \item \textbf{Consonants (core):} m n (ng), l r, v s z, š (sh), h, p b, t d, k g, f, j (y) \\
        (optional: th, q in ritual/maths)
  \item \textbf{Stress:} penultimate syllable of the phonological word; \textbf{particles unstressed}.
\end{itemize}

% =======================
\section*{Word Formation}
\textbf{Head-final compounding:} the \textbf{last root} is the semantic head.
\begin{itemize}
  \item \textbf{te-log} = truth-reason = proof / theorem-ish
  \item \textbf{sae-mu} = sorrow-memory = mourning/nostalgia nuance
  \item \textbf{an-xo-mi} = love-bind-meaning = vow/affective bond-word
\end{itemize}

\textbf{Category hints by head:}
\begin{itemize}
  \item ends in \textbf{es} states/being; \textbf{ki} actions; \textbf{mi} meaning/symbol
  \item ends in \textbf{log} reasoning/system; \textbf{xo} bonds/relations
  \item ends in \textbf{set} collectives; \textbf{ord} sequences/order
\end{itemize}

\textbf{Compression (historical erosion):} frequent compounds may shorten over time (dialects).

% =======================
\section*{Core Syntax}
\textbf{Main clauses (V2):}\\
Exactly one element is fronted; finite verb is \textbf{second}.
\begin{itemize}
  \item \exm{nuu ya vi-te ri} \glt{Now I witness it.}
  \item \exm{an ya vi-te ri} \glt{Lovingly, I witness it.}
  \item \exm{mu-sae-in ya vi-te ri} \glt{Within sorrow-memory, I witness it.}
\end{itemize}

\textbf{Subordinate clauses (verb-final):}\\
A subordinator triggers verb-last order (define subordinators as needed in your canon).

% =======================
\section*{Questions}
No inversion. Use \textbf{qe} (uncertainty) as interrogative mood/marker.
\begin{itemize}
  \item \exm{qe tu vi-te} \glt{Do you see the truth?}
  \item \exm{qe pos we xo-sol} \glt{Will we bind the vow?}
\end{itemize}

% =======================
\section*{Negation \& Polarity}
Negator: \textbf{na}. Only one negator per clause (unless poetic/paradox).
\begin{itemize}
  \item \textbf{Clause negation:} \exm{ya na vi-te} \glt{I do not see truth.}
  \item \textbf{Constituent negation:} \exm{ya vi na-te} \glt{I see not-truth (falsehood).}
\end{itemize}

\textbf{Strong affirmation:} front \textbf{te} (truth).
\begin{itemize}
  \item \exm{te ya vi} \glt{Truly, I see.}
\end{itemize}

% =======================
\section*{TAM: Mood--Tense--Aspect Order (Sacred)}
\textbf{Order is strict:} \textbf{[MOOD] [TENSE] [ASPECT] VERB}

\subsection*{Tense (Spanish-like)}
\begin{tabularx}{\linewidth}{@{}>{\bfseries}l X@{}}
\textbf{0} & present \\
\textbf{pra} & past (preterite/completed) \\
\textbf{dur-pra} & imperfect past (ongoing then) \\
\textbf{pos} & future \\
\textbf{pos-pra} & conditional (would) \\
\textbf{ret-pra} & perfect (have done) \\
\end{tabularx}

\subsection*{Aspect}
\begin{tabularx}{\linewidth}{@{}>{\bfseries}l X@{}}
\textbf{pun} & punctual/instant \\
\textbf{dur} & continuous \\
\textbf{vek} & change-of-state / become \\
\textbf{ret} & iterative/again \\
\end{tabularx}

\subsection*{Mood}
\begin{tabularx}{\linewidth}{@{}>{\bfseries}l X@{}}
\textbf{0} & indicative (reality) \\
\textbf{qe} & subjunctive / uncertain / imagined \\
\textbf{dar} & imperative / must \\
\textbf{fre} & permissive / may \\
\textbf{ve} & awe/fear mood (reverent) \\
\textbf{an} & loving volition / benevolent intent \\
\end{tabularx}

\subsection*{TAM Examples}
\begin{itemize}
  \item \exm{ya vi-te} \glt{I witness / I see truth.}
  \item \exm{ya pra vi-te} \glt{I witnessed.}
  \item \exm{ya dur-pra vi-te} \glt{I was witnessing.}
  \item \exm{ya pos vi-te} \glt{I will witness.}
  \item \exm{ya pos-pra vi-te} \glt{I would witness.}
  \item \exm{ya ret-pra vi-te} \glt{I have witnessed.}
  \item \exm{qe pos ya vi-te} \glt{That I might see.}
  \item \exm{dar ki} \glt{Act.}
\end{itemize}

\begin{itemize}
  \item \exm{te-log-xo} \glt{by means of proof}
  \item \exm{mu-sae-in} \glt{within sorrow-memory}
  \item \exm{ya-we} \glt{with me}
  \item \exm{an-na} \glt{without love / lovelessly}
\end{itemize}

% =======================
\section*{Pronouns (minimal)}
\begin{tabularx}{\linewidth}{@{}>{\bfseries}l X@{}}
ya & I / speaker-self \\
tu & you (context decides sg/pl) \\
we & we / collective \\
un & one / generic agent / self \\
du & other / counterpart / they (contextual) \\
\end{tabularx}

Pronouns decline like nouns: \exm{ya-ri} (me), \exm{tu-ha} (to you), etc.\\
\textbf{Pro-drop:} omit pronouns when obvious.

% =======================
\section*{Registers}
\begin{itemize}
  \item \textbf{High Ordic:} no pronouns; timeless; strict order; cannot casually lie.
  \item \textbf{Middle Ordic:} administration/ritual; compressed compounds; partial erosion.
  \item \textbf{Low Ordic:} living speech; particles drop; emotion leaks; slang emerges.
\end{itemize}

% =======================
\section*{Performatives (speech that acts)}
Performative classes (core roots):
\begin{itemize}
  \item \textbf{Binding:} xo (e.g., \exm{an xo-sol} vow bound in love)
  \item \textbf{Release:} sha + gra (e.g., \exm{gra sha-xo} mercy unbinds)
  \item \textbf{Naming:} so-mi-es (meaning spoken into being)
  \item \textbf{Exile:} ex-ord (placed outside order)
  \item \textbf{Witness:} vi-te (truth-locking)
\end{itemize}

\section*{Canon: Ord-ret (short excerpt)}
\begin{tabularx}{\linewidth}{@{}l X@{}}

\exm{te ord es.} & Truth: order is. \\
\exm{pra zen es-na.} & Before, void was absent. \\
\exm{an xo dur.} & Love binds continuously. \\
\exm{qe ir vek.} & Doubt: anger becomes. \\

\end{tabularx}

\section*{Selected Root Clusters (for quick building)}
\textbf{Ontology:} es (be), na (not/absence), un (one), du (other), al (all), par (part), ka (form), il (essence)\\
\textbf{Time:} nuu (now), pra (before/past), pos (after/future), dur (duration), pun (instant), ret (again/return), sim (while), vek (become)\\
\textbf{Space:} in (inside), ex (outside), su (above), da (below), tri (through), ne (near), far (far), lin (path/toward)\\
\textbf{Force:} ki (do), se (allow), ta (cause), vu (resist), ha (give), ri (take), xo (bind), sha (sever)\\
\textbf{Mind:} me (mind), vi (see), lu (know), mu (remember), qe (doubt), so (say), te (truth), mi (meaning)\\
\textbf{Emotion:} an (love), ir (anger), sae (sorrow), jo (joy), ve (awe/fear), shael (calm), ru (desire), ny (shame)\\
\textbf{Social:} we (we), tu (you), ya (I), dar (must), fre (may), sol (vow), jur (law), gra (grace)\\
\textbf{Math/Structure:} zen (zero), sum (add), dif (diff), mul (multiply), log (reason), set (set), eq (equal), ord (order)

\section*{Quick Build Recipes}
\begin{itemize}
  \item \textbf{Proof/theorem:} te-log (truth-reason)
  \item \textbf{Witness/verify:} vi-te (see-truth)
  \item \textbf{Nostalgia:} mu-sae (memory-sorrow)
  \item \textbf{Bonded promise:} an-xo-sol (love-bind-vow)
  \item \textbf{Exile state:} ex-ord (outside-order)
  \item \textbf{System/order-of-sets:} set-ord / ord-set (choose head for nuance)
\end{itemize}

% =======================
\section*{Ord-Gate (artifact) Activation}
A canonical access utterance:
\begin{itemize}
  \item \exm{gra xo we ord} \glt{Mercy binds the collective order.}
  \item Constraints: no coercive mood (\exm{dar} kills it), sincerity required.
\end{itemize}

% =======================
\section*{Sha-Te (artifact) Summary}
\begin{itemize}
  \item Weapon inscription: \exm{te vi sha} \glt{Truth witnessed through severance.}
  \item Refuses unjustified force; may render wielder unable to use \exm{te} if abused.
\end{itemize}
